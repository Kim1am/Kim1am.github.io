  创建一个实例很简单，然后我们可以使用constructor构造函数为我们这个实例挂载一些属性,全部挂载到当前实例的this上

class BarrageCanvas {
    constructor(cDom, vDom, options = {}) {
      // 建议版判断是否传入了DOM节点
      if (!cDom || !vDom) return
      this.canvas = cDom
      this.video = vDom
      this.canvas.width = vDom.width
      this.canvas.height = vDom.height
      // 上下文
      this.ctx = cDom.getContext('2d')
      // 弹幕默认参数
      let defaultOpt = {
        color: '#fff',
        speed: 1,
        fontSize: 20,
        data: []
      }
      Object.assign(this, defaultOpt, options)
      //控制位，因为视频会有暂停播放状态，如果暂停就暂停渲染弹幕
      this.isPaused = true
      this.barrages = this.data.map(item => new Barrages(item, this))
      this.render()
    }
    render (){
    //  绘制弹幕操作
    }
}

  其实开头判断不够严谨，还需要判断一下tagName是否真的就是相对应的canvas和video。然后这里也是用了ES6的默认值。另外可以设置弹幕的参数默认值。 而Object.assign(this, defaultOpt, options);的作用是把弹幕数据options和default一起挂载到this当前实例上面。 Barrages类看名字就可以知道专职生成弹幕实体的，方便我们以后的用render去渲染弹幕。

render函数并不复杂，粗略来讲就是不断递归去“绘画”，然后给我们一种移动的感觉，但是不是用setTimeOut或者setInterval这种去弄，因为通过这种方式去弄，会出现卡顿，动画不流畅的情况，这里使用一个独特的API window.requestAnimationFrame，其实大概就是让浏览器去循环递归执行我们的函数。而且每一次render都需要先把上一次画布的清除掉，否则将一直存在于画布上

class BarrageCanvas {
    //………………
    render (){
        // 每次渲染弹幕都要清除上一次的画布
        this.clear()
        //重点方法 渲染弹幕
        this.renderBarrage()
        // 如果没有暂停就继续渲染弹幕
        if (this.isPaused === false) {
          // 注意这里的递归this，不改变this指向的话指向domcument
          requestAnimationFrame(this.render.bind(this))
        }
    }
    clear() {
      // 清除整块画布
       this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height)
    }
}

# Barrages类--弹幕加工厂

Barrages类本身只专责负责一个功能，就是处理每条弹幕barrage属性（字号、颜色、速度、宽度（这个重点），出现时间） 以供BarrageCanvas类渲染弹幕，常规的还是写一个constructor构造函数去挂载属性。

class Barrages {
  constructor(barrageObj, bcCTX) {
    // 弹幕文本
    this.txt = barrageObj.value
    // 弹幕出现时间
    this.time = barrageObj.time
    // 弹幕数据
    this.barrageObj = barrageObj
    // BarrageCanvas实例内容
    this.bcCTX = bcCTX
  }
}

  之后就是弹幕制作中的其中一个难点，就是确定弹幕在canvas上的出现位置，消失位置。而且还要获取弹幕本身的width，为什么要获取宽度，用于计算弹幕“走完”时机，详细看图。

  同理，也会出现Y轴溢出情况，但是Y轴的坐标永远固定，所以我们只需要在弹幕初始化的确定好Y轴出现点，在超出Y轴的情况，我们可以选择“拉回”显示空间。看图

  把这些弄清楚后基本上这个弹幕功能也完成了七成左右了，其余剩下得无非就是颜色、字号、速度、出现X轴Y轴坐标，宽度那些的初始化而已。 render函数就是一个“写字”过程，把弹幕“写”到canvas上。

class Barrages {
 //……
   init() {
     // 如果弹幕data没有对应属性，则取default里面的配置
     this.color = this.barrageObj.color || this.bcCTX.color
     this.speed = this.barrageObj.speed || this.bcCTX.speed
     this.fontSize = this.barrageObj.fontSize || this.bcCTX.fontSize
 
     // 获取弹幕文本的对应宽以计算对应弹幕的“离场位置”
     let pDOM = document.createElement('span');
     pDOM.innerText = this.txt
     pDOM.style.fontSize = this.fontSize + 'px'
     pDOM.style.position = 'absolute'
     document.body.appendChild(pDOM)
 
     this.width = pDOM.clientWidth
     document.body.removeChild(pDOM)
     // 弹幕的X轴出现 位置。应该在canvas的最右端
     this.showX = this.bcCTX.canvas.width
     // 弹幕的Y轴出现位置，随机
     this.showY = this.bcCTX.canvas.height * Math.random()
     // 因为弹幕是从左上角开始绘制，所以当弹幕出现y轴位置小于本身弹幕高度即显示不完整，须控制
     if (this.showY < this.fontSize) {
       this.showY = this.fontSize
     } else if (this.Y > this.bcCTX.canvas.height - this.fontSize) {
       // 当弹幕“溢出”Y轴范围时，需要拉回范围内
       this.showY = this.bcCTX.canvas.height - this.fontSize;
     }
   }
   //渲染弹幕函数
   render() {
     this.bcCTX.ctx.font = `${this.fontSize}px Arial`;
     this.bcCTX.ctx.fillStyle = this.color
     this.bcCTX.ctx.fillText(this.txt, this.showX, this.showY);
   }
}

# renderBarrage --弹幕控制中枢

  最后剩下关键的一个地方了，控制弹幕的绘画功能都写在这个方法里面。其实内涵的功能如下：弹幕时间判断，弹幕初始化（init），弹幕移动控制,X轴溢出位置判断。 每个弹幕都要经过Barrage类初始化一次，然后记录一个表示位为true，确定好弹幕的起始点那些就行。而X轴溢出就根据图二的原理去计算就行。

class CanvasBarrage {
  // …………
    renderBarrage() {
      let currentTime = this.video.currentTime
      this.barrages.forEach(barrage => {
        // 判断弹幕是否在这个时间点出现，并且是否已经渲染过弹幕,true的时候代表已经离开video位置
        if (currentTime >= barrage.time && !barrage.outRange) {
          if (!barrage.isInit) {
            // 初始化为弹幕确定颜色，大小，出现位置（x,y），位置
            barrage.init()
            barrage.isInit = true
          }
  
          //弹幕移动
          barrage.showX = barrage.showX - barrage.speed
          barrage.render()
  
          // 判断是否已经离开可视范围，即超出video宽度+本身弹幕宽度
          if (barrage.showX < -barrage.width) {
            barrage.outRange = true
          }
        }
      });
    }
}

# 与播放器联动

  前面写了那么多，都没有试过综合视频是怎么使用，最先做的肯定是addEventListner去监听是否播放，还有暂停。当我们开始播放的时候就把isPaused位置 为false,证明开始播放，同时调用render函数去开始画弹幕，相反暂停则置反。

videoDOM.addEventListener('play', () => {
  barrageCanvas.isPaused = false
  barrageCanvas.render()
})
videoDOM.addEventListener('pause', () => {
  barrageCanvas.isPaused = true
})

  但是如果我们拖动进度条呢？这个时候首先我们做的肯定是清空画布，然后把根据当前播放器的事件一一对比每条弹幕出现的时间，当当前播放时间小于或等于弹幕出现时间，则应当出现 当然，之前我们弄X轴弹幕溢出的时候，outRange属性记得重置为false,这样才能支持往回拖，弹幕重新出来的操作。

// 拖动
videoDOM.addEventListener('seeked', () => {
  barrageCanvas.reset()
})
class CanvasBarrage {
  //………………
    seeked() {
      this.clear()
      let currentTime = this.video.currentTime
      this.barrages.forEach(barrage => {
        // 设定弹幕没有超出范围
        barrage.outRange = false
        // 当弹幕出现时间比当前时间大或者等于则重置
        if (currentTime <= barrage.time) {
          // 就把isInit重设为false，这样才会重新初始化渲染
          barrage.isInit = false;
        } else { // 其他时间对比不匹配的，flag还是true不用重新渲染
          barrage.outRange = true;
        }
      });
    }
}

  最后，就是发送弹幕了，这个就不做过多解释了，直接add一个弹幕对象就OK。直接上代码

// 发送
btn.addEventListener('click', send)

function send() {
  let value = bTxt.value
  let time = videoDOM.currentTime
  let color = bColor.value
  let fontSize = bSize.value
  let obj = {
    value,
    time,
    color,
    fontSize
  }
  // 添加弹幕数据
  barrageCanvas.add(obj);
  bTxt.value = ''
}
class CanvasBarrage {
  //………………
  add(obj) {
    // 实际上就是往barrages数组里再添加一项Barrage的实例而已
    this.barrages.push(new Barrages(obj, this));
  }
}
